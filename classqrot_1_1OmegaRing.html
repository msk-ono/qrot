Detailed Description

template<RealRingConcept Ring>
class qrot::OmegaRing< Ring >

RealRing + omega * RealRing + omega^2 * RealRing + omega^3 * RealRing.

omega = \sqrt{1/2} + \sqrt{1/2} i (omega^4=-1) x_[0] + x_[1] * omega + x_[2] * omega^2 + x_[3] * omega^3
